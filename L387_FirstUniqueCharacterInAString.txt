class Solution {
    public int firstUniqChar(String s) {
        for (int i = 0; i < s.length(); i++) {
            if (!(s.substring(0, i).contains(Character.toString(s.charAt(i)))
                    || s.substring(i + 1, s.length()).contains(Character.toString(s.charAt(i))))) {
                return i;
            }

        }
        return -1;
    }
}
