class Solution {
    public int mySqrt(int x) {
        int j = 0;

        for (int i = 1; i <= x; i++) {

            if (x == 0 || x == 1) {
                return x;
            }
            if (x / i > i) {
                continue;
            }
            if ((x / i < i) || (x / i == i)) {
                j = x / i;
            }
            break;
        }
        return j;
    }
}
