class Solution {
    public boolean isPalindrome(int x) {
     StringBuilder builder = new StringBuilder();
        String s = Integer.toString(x);
        int len = s.length();


        for(int i=len-1;i>=0;i--){
builder.append(s.charAt(i));

        }
        String reversed = builder.toString();
        if(reversed.equals(s))
             return true;
        else return false;

    }
    
}
